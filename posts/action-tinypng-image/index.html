    #!/usr/bin/sh
if [ $# != 1 ] ; then
	echo "USAGE: $0 [Image file path]"
	echo " e.g.: $0 ~/download/rad.png"
	exit 1;
fi
filename="$1"
nfname="`date '+%Y%m%d%H%M%S'`.${filename#*.}"
echo "![](https://cdn.jsdelivr.net/gh/user/cdn/img/$nfname)"
cd ./cdn
git pull
cd ..
cp "$filename" "./cdn/img/$nfname.O"
cd ./cdn
git add -A
git push
cd ..
